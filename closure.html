Closure Function

Closures are functions inside a function and it makes a normal function stateful. It is used to create a self-contained state to avoid global variables.

Sample Code: 

        function closureFunction() {
          var x = 0;

          function Increment() {
            document.getElementById("demo").innerHTML =
              "Output : counter: " + ++x;
          }

          function getXvalue() {
            return x;
          }

          function init() {
            return (x = 0);
          }

          return {
            Increment,
            getXvalue,
            init,
          };
        }

        var ref = closureFunction();
        ref.Increment();
        ref.Increment();
        ref.Increment();
        ref.getXvalue();
        ref.init();
